using Finisar.SQLite;
            
string MyConString = "Data Source=Demo.db3;Version=3;New=True;Compress=True;";	

SQLiteConnection sql_con = new SQLiteConnection();
sql_con.Open();

SQLiteCommand sql_cmd =	sql_con.CreateCommand();

sql_cmd.CommandText="create table t1(id integer, descr varchar2(40))";
sql_cmd.ExecuteNonQuery(); 

sql_cmd.CommandText="select * from t1";
SQLiteDataReader reader = sql_cmd.ExecuteReader();

while (reader.Read())
{
	Console.WriteLine("{0}, {1}", reader[0].ToString(), reader[1].ToString());	
}

reader.Close();
reader.Dispose();
 
sql_cmd.Dispose();

sql_con.Close();

Console.Write("Press any key to continue . . . ");
Console.ReadKey(true);
